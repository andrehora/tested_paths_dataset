16    :return: The rendered value for the default variable.
17    """
18    if raw is None or isinstance(raw, bool):
19        return raw
20    elif isinstance(raw, dict):
21        return {

23                env, v, cookiecutter_dict
24            )
25            for k, v in raw.items()
26        }
27    elif isinstance(raw, list):
28        return [render_variable(env, v, cookiecutter_dict) for v in raw]
29    elif not isinstance(raw, str):
30        raw = str(raw)

34    return template.render(cookiecutter=cookiecutter_dict)
