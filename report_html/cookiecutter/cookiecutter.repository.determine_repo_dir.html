16
17    :param template: A directory containing a project template directory,
18        or a URL to a git repository.
19    :param abbreviations: A dictionary of repository abbreviation
20        definitions.
21    :param clone_to_dir: The directory to clone the repository into.
22    :param checkout: The branch, tag or commit ID to checkout after clone.
23    :param no_input: Do not prompt for user input and eventually force a refresh of
24        cached resources.
25    :param password: The password to use when extracting the repository.
26    :param directory: Directory within repo where cookiecutter.json lives.
27    :return: A tuple containing the cookiecutter template directory, and
28        a boolean describing whether that directory should be cleaned up
29        after the template has been instantiated.
30    :raises: `RepositoryNotFound` if a repository directory could not be found.
31    """
32    template = expand_abbreviations(template, abbreviations)
33
34    if is_zip_file(template):
35        unzipped_dir = unzip(
36            zip_uri=template,
37            is_url=is_repo_url(template),
38            clone_to_dir=clone_to_dir,
39            no_input=no_input,
40            password=password,
41        )
42        repository_candidates = [unzipped_dir]
43        cleanup = True
44    elif is_repo_url(template):
45        cloned_repo = clone(
46            repo_url=template,
47            checkout=checkout,
48            clone_to_dir=clone_to_dir,
49            no_input=no_input,
50        )
51        repository_candidates = [cloned_repo]
52        cleanup = False
53    else:
54        repository_candidates = [template, os.path.join(clone_to_dir, template)]
55        cleanup = False
56
57    if directory:
58        repository_candidates = [
59            os.path.join(s, directory) for s in repository_candidates
60        ]
61
62    for repo_candidate in repository_candidates:
63        if repository_has_cookiecutter_json(repo_candidate):
64            return repo_candidate, cleanup
65
66    raise RepositoryNotFound(
67        'A valid repository for "{}" could not be found in the following '
68        'locations:\n{}'.format(template, '\n'.join(repository_candidates))
69    )
