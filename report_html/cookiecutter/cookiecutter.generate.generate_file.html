9
10    Precondition:
11
12        When calling `generate_file()`, the root template dir must be the
13        current working directory. Using `utils.work_in()` is the recommended
14        way to perform this directory change.
15
16    :param project_dir: Absolute path to the resulting generated project.
17    :param infile: Input file to generate the file from. Relative to the root
18        template dir.
19    :param context: Dict for populating the cookiecutter's variables.
20    :param env: Jinja2 template execution environment.
21    """
22    logger.debug('Processing file %s', infile)
23
24    # Render the path to the output file (not including the root project dir)
25    outfile_tmpl = env.from_string(infile)
26
27    outfile = os.path.join(project_dir, outfile_tmpl.render(**context))
28    file_name_is_empty = os.path.isdir(outfile)
29    if file_name_is_empty:
30        logger.debug('The resulting file name is empty: %s', outfile)
31        return
32
33    if skip_if_file_exists and os.path.exists(outfile):
34        logger.debug('The resulting file already exists: %s', outfile)
35        return
36
37    logger.debug('Created file at %s', outfile)
38
39    # Just copy over binary files. Don't render.
40    logger.debug("Check %s to see if it's a binary", infile)
41    if is_binary(infile):
42        logger.debug('Copying binary %s to %s without rendering', infile, outfile)
43        shutil.copyfile(infile, outfile)
44    else:
45        # Force fwd slashes on Windows for get_template
46        # This is a by-design Jinja issue
47        infile_fwd_slashes = infile.replace(os.path.sep, '/')
48
49        # Render the file
50        try:
51            tmpl = env.get_template(infile_fwd_slashes)
52        except TemplateSyntaxError as exception:
53            # Disable translated so that printed exception contains verbose
54            # information about syntax error location
55            exception.translated = False
56            raise
57        rendered_file = tmpl.render(**context)
58
59        # Detect original file newline to output the rendered file
60        # note: newline='' ensures newlines are not converted
61        with open(infile, encoding='utf-8', newline='') as rd:
62            rd.readline()  # Read the first line to load 'newlines' value
63
64            # Use `_new_lines` overwrite from context, if configured.
65            newline = rd.newlines
66            if context['cookiecutter'].get('_new_lines', False):
67                newline = context['cookiecutter']['_new_lines']
68                logger.debug('Overwriting end line character with %s', newline)
69
70        logger.debug('Writing contents to file %s', outfile)
71
72        with open(outfile, 'w', encoding='utf-8', newline=newline) as fh:
73            fh.write(rendered_file)
74
75    # Apply file permissions to output file
76    shutil.copymode(infile, outfile)

3
4    Dealing with infile appropriately:
5
6        a. If infile is a binary file, copy it over without rendering.
7        b. If infile is a text file, render its contents and write the
8           rendered infile to outfile.
9
10    Precondition:
11
12        When calling `generate_file()`, the root template dir must be the
13        current working directory. Using `utils.work_in()` is the recommended
14        way to perform this directory change.
15
16    :param project_dir: Absolute path to the resulting generated project.
17    :param infile: Input file to generate the file from. Relative to the root
18        template dir.
19    :param context: Dict for populating the cookiecutter's variables.
20    :param env: Jinja2 template execution environment.
21    """
22    logger.debug('Processing file %s', infile)
23
24    # Render the path to the output file (not including the root project dir)
25    outfile_tmpl = env.from_string(infile)
26
27    outfile = os.path.join(project_dir, outfile_tmpl.render(**context))
28    file_name_is_empty = os.path.isdir(outfile)
29    if file_name_is_empty:
30        logger.debug('The resulting file name is empty: %s', outfile)
31        return
32
33    if skip_if_file_exists and os.path.exists(outfile):
34        logger.debug('The resulting file already exists: %s', outfile)
35        return
36
37    logger.debug('Created file at %s', outfile)
38
39    # Just copy over binary files. Don't render.
40    logger.debug("Check %s to see if it's a binary", infile)
41    if is_binary(infile):
42        logger.debug('Copying binary %s to %s without rendering', infile, outfile)
43        shutil.copyfile(infile, outfile)
44    else:
45        # Force fwd slashes on Windows for get_template
46        # This is a by-design Jinja issue
47        infile_fwd_slashes = infile.replace(os.path.sep, '/')
48
49        # Render the file
50        try:
51            tmpl = env.get_template(infile_fwd_slashes)

58
59        # Detect original file newline to output the rendered file

62            rd.readline()  # Read the first line to load 'newlines' value
63
64            # Use `_new_lines` overwrite from context, if configured.

66            if context['cookiecutter'].get('_new_lines', False):

68                logger.debug('Overwriting end line character with %s', newline)
69
70        logger.debug('Writing contents to file %s', outfile)
71
72        with open(outfile, 'w', encoding='utf-8', newline=newline) as fh:
73            fh.write(rendered_file)
74
75    # Apply file permissions to output file
76    shutil.copymode(infile, outfile)
