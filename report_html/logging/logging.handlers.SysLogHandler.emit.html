2        """
3        Emit a record.
4
5        The record is formatted, and then sent to the syslog server. If
6        exception information is present, it is NOT sent to the server.
7        """
8        try:
9            msg = self.format(record)
10            if self.ident:
11                msg = self.ident + msg
12            if self.append_nul:
13                msg += '\000'
14
15            # We need to convert record level to lowercase, maybe this will
16            # change in the future.
17            prio = '<%d>' % self.encodePriority(self.facility,
18                                                self.mapPriority(record.levelname))
19            prio = prio.encode('utf-8')
20            # Message is a string. Convert to bytes as required by RFC 5424
21            msg = msg.encode('utf-8')
22            msg = prio + msg
23            if self.unixsocket:
24                try:
25                    self.socket.send(msg)
26                except OSError:
27                    self.socket.close()
28                    self._connect_unixsocket(self.address)
29                    self.socket.send(msg)
30            elif self.socktype == socket.SOCK_DGRAM:
31                self.socket.sendto(msg, self.address)
32            else:
33                self.socket.sendall(msg)
34        except Exception:
35            self.handleError(record)
