15            klass = eval(klass, vars(logging))
16        except (AttributeError, NameError):
17            klass = _resolve(klass)
18        args = section.get("args", '()')
19        args = eval(args, vars(logging))
20        kwargs = section.get("kwargs", '{}')

22        h = klass(*args, **kwargs)
23        h.name = hand
24        if "level" in section:
25            level = section["level"]
26            h.setLevel(level)
27        if len(fmt):
28            h.setFormatter(formatters[fmt])
29        if issubclass(klass, logging.handlers.MemoryHandler):
30            target = section.get("target", "")
31            if len(target): #the target handler may not be loaded yet, so keep for later...
32                fixups.append((h, target))
33        handlers[hand] = h
34    #now all handlers are loaded, fixup inter-handler references...
35    for h, t in fixups:
36        h.setTarget(handlers[t])
37    return handlers
