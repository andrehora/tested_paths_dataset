6        # If we are rolling over at midnight or weekly, then the interval is already known.
7        # What we need to figure out is WHEN the next interval is.  In other words,
8        # if you are rolling over at midnight, then your base interval is 1 day,
9        # but you want to start that one day clock at midnight, not now.  So, we
10        # have to fudge the rolloverAt value in order to trigger the first rollover
11        # at the right time.  After that, the regular interval will take care of
12        # the rest.  Note that this code doesn't care about leap seconds. :)
13        if self.when == 'MIDNIGHT' or self.when.startswith('W'):
14            # This could be done with less code, but I wanted it to be clear
15            if self.utc:
16                t = time.gmtime(currentTime)
17            else:
18                t = time.localtime(currentTime)
19            currentHour = t[3]
20            currentMinute = t[4]
21            currentSecond = t[5]
22            currentDay = t[6]
23            # r is the number of seconds left between now and the next rotation
24            if self.atTime is None:
25                rotate_ts = _MIDNIGHT
26            else:
27                rotate_ts = ((self.atTime.hour * 60 + self.atTime.minute)*60 +
28                    self.atTime.second)
29
30            r = rotate_ts - ((currentHour * 60 + currentMinute) * 60 +
31                currentSecond)
32            if r < 0:
33                # Rotate time is before the current time (for example when
34                # self.rotateAt is 13:45 and it now 14:15), rotation is
35                # tomorrow.
36                r += _MIDNIGHT
37                currentDay = (currentDay + 1) % 7
38            result = currentTime + r
39            # If we are rolling over on a certain day, add in the number of days until
40            # the next rollover, but offset by 1 since we just calculated the time
41            # until the next day starts.  There are three cases:
42            # Case 1) The day to rollover is today; in this case, do nothing
43            # Case 2) The day to rollover is further in the interval (i.e., today is
44            #         day 2 (Wednesday) and rollover is on day 6 (Sunday).  Days to
45            #         next rollover is simply 6 - 2 - 1, or 3.
46            # Case 3) The day to rollover is behind us in the interval (i.e., today
47            #         is day 5 (Saturday) and rollover is on day 3 (Thursday).
48            #         Days to rollover is 6 - 5 + 3, or 4.  In this case, it's the
49            #         number of days left in the current week (1) plus the number
50            #         of days in the next week until the rollover day (3).
51            # The calculations described in 2) and 3) above need to have a day added.
52            # This is because the above time calculation takes us to midnight on this
53            # day, i.e. the start of the next day.
54            if self.when.startswith('W'):
55                day = currentDay # 0 is Monday
56                if day != self.dayOfWeek:
57                    if day < self.dayOfWeek:
58                        daysToWait = self.dayOfWeek - day
59                    else:
60                        daysToWait = 6 - day + self.dayOfWeek + 1
61                    newRolloverAt = result + (daysToWait * (60 * 60 * 24))
62                    if not self.utc:
63                        dstNow = t[-1]
64                        dstAtRollover = time.localtime(newRolloverAt)[-1]
65                        if dstNow != dstAtRollover:

67                                addend = -3600
68                            else:           # DST bows out before next rollover, so we need to add an hour
69                                addend = 3600
70                            newRolloverAt += addend
71                    result = newRolloverAt
72        return result
