13            fileno = o.fileno()
14        except (io.UnsupportedOperation, AttributeError):
15            # AttributeError is a surprising exception, seeing as how we've just checked
16            # that `hasattr(o, 'fileno')`.  It happens for objects obtained via
17            # `Tarfile.extractfile()`, per issue 5229.
18            pass
19        else:
20            total_length = os.fstat(fileno).st_size
21
22            # Having used fstat to determine the file length, we need to
23            # confirm that this file was opened up in binary mode.
24            if "b" not in o.mode:
25                warnings.warn(
26                    (
27                        "Requests has determined the content-length for this "
28                        "request using the binary size of the file: however, the "
29                        "file has been opened in text mode (i.e. without the 'b' "
30                        "flag in the mode). This may lead to an incorrect "
31                        "content-length. In Requests 3.0, support will be removed "
32                        "for files in text mode."
33                    ),
34                    FileModeWarning,
35                )
36
37    if hasattr(o, "tell"):
38        try:
39            current_position = o.tell()
40        except OSError:
41            # This can happen in some weird situations, such as when the file
42            # is actually a special file descriptor like stdin. In this
43            # instance, we don't know what the length is, so set it to zero and
44            # let requests chunk it instead.
45            if total_length is not None:
46                current_position = total_length
47        else:
48            if hasattr(o, "seek") and total_length is None:
49                # StringIO and BytesIO have seek but no usable fileno
50                try:
51                    # seek to end of file
52                    o.seek(0, 2)
53                    total_length = o.tell()
54
55                    # seek back to current position to support
56                    # partially read file-like objects
57                    o.seek(current_position or 0)
58                except OSError:
59                    total_length = 0
60
61    if total_length is None:
62        total_length = 0
63
64    return max(0, total_length - current_position)
