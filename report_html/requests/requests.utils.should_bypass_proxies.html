6    """
7    # Prioritize lowercase environment variables over uppercase
8    # to keep a consistent behaviour with other http projects (curl, wget).
9    def get_proxy(key):
10        return os.environ.get(key) or os.environ.get(key.upper())
11
12    # First check whether no_proxy is defined. If it is, check that the URL
13    # we're getting isn't in the no_proxy list.
14    no_proxy_arg = no_proxy
15    if no_proxy is None:
16        no_proxy = get_proxy("no_proxy")
17    parsed = urlparse(url)
18
19    if parsed.hostname is None:
20        # URLs don't always have hostnames, e.g. file:/// urls.
21        return True
22
23    if no_proxy:
24        # We need to check whether we match here. We need to see if we match
25        # the end of the hostname, both with and without the port.
26        no_proxy = (host for host in no_proxy.replace(" ", "").split(",") if host)
27
28        if is_ipv4_address(parsed.hostname):
29            for proxy_ip in no_proxy:
30                if is_valid_cidr(proxy_ip):
31                    if address_in_network(parsed.hostname, proxy_ip):
32                        return True
33                elif parsed.hostname == proxy_ip:
34                    # If no_proxy ip was defined in plain IP notation instead of cidr notation &
35                    # matches the IP of the index
36                    return True
37        else:
38            host_with_port = parsed.hostname
39            if parsed.port:
40                host_with_port += f":{parsed.port}"
41
42            for host in no_proxy:
43                if parsed.hostname.endswith(host) or host_with_port.endswith(host):
44                    # The URL does match something in no_proxy, so we don't want
45                    # to apply the proxies on this URL.
46                    return True
47
48    with set_environ("no_proxy", no_proxy_arg):
