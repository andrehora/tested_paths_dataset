8        :param verify: Either a boolean, in which case it controls whether we verify

11        :param cert: The SSL certificate to verify.
12        """
13        if url.lower().startswith("https") and verify:
14
15            cert_loc = None
16
17            # Allow self-specified cert location.
18            if verify is not True:
19                cert_loc = verify
20
21            if not cert_loc:
22                cert_loc = extract_zipped_paths(DEFAULT_CA_BUNDLE_PATH)
23
24            if not cert_loc or not os.path.exists(cert_loc):

29
