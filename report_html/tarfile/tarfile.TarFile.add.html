31        if filter is not None:
32            tarinfo = filter(tarinfo)
33            if tarinfo is None:
34                self._dbg(2, "tarfile: Excluded %r" % name)

36
37        # Append the tar header and data to the archive.
38        if tarinfo.isreg():
39            with bltn_open(name, "rb") as f:
40                self.addfile(tarinfo, f)
41
42        elif tarinfo.isdir():
43            self.addfile(tarinfo)
44            if recursive:
45                for f in sorted(os.listdir(name)):
46                    self.add(os.path.join(name, f), os.path.join(arcname, f),
47                            recursive, filter=filter)
48
49        else:
50            self.addfile(tarinfo)
