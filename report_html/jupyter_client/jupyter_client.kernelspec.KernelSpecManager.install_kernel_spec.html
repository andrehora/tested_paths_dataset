1def install_kernel_spec(
2        self, source_dir, kernel_name=None, user=False, replace=None, prefix=None
3    ):
4        """Install a kernel spec by copying its directory.
5
6        If ``kernel_name`` is not given, the basename of ``source_dir`` will
7        be used.
8
9        If ``user`` is False, it will attempt to install into the systemwide
10        kernel registry. If the process does not have appropriate permissions,
11        an :exc:`OSError` will be raised.
12
13        If ``prefix`` is given, the kernelspec will be installed to
14        PREFIX/share/jupyter/kernels/KERNEL_NAME. This can be sys.prefix
15        for installation inside virtual or conda envs.
16        """
17        source_dir = source_dir.rstrip("/\\")
18        if not kernel_name:
19            kernel_name = os.path.basename(source_dir)
20        kernel_name = kernel_name.lower()
21        if not _is_valid_kernel_name(kernel_name):
22            msg = f"Invalid kernel name {kernel_name!r}.  {_kernel_name_description}"
23            raise ValueError(msg)
24
25        if user and prefix:
26            msg = "Can't specify both user and prefix. Please choose one or the other."
27            raise ValueError(msg)
28
29        if replace is not None:
30            warnings.warn(
31                "replace is ignored. Installing a kernelspec always replaces an existing "
32                "installation",
33                DeprecationWarning,
34                stacklevel=2,
35            )
36
37        destination = self._get_destination_dir(kernel_name, user=user, prefix=prefix)
38        self.log.debug("Installing kernelspec in %s", destination)
39
40        kernel_dir = os.path.dirname(destination)
41        if kernel_dir not in self.kernel_dirs:
42            self.log.warning(
43                "Installing to %s, which is not in %s. The kernelspec may not be found.",
44                kernel_dir,
45                self.kernel_dirs,
46            )
47
48        if os.path.isdir(destination):
49            self.log.info("Removing existing kernelspec in %s", destination)
50            shutil.rmtree(destination)
51
52        shutil.copytree(source_dir, destination)
53        self.log.info("Installed kernelspec %s in %s", kernel_name, destination)
54        return destination
