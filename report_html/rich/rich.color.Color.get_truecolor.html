4        """Get an equivalent color triplet for this color.
5
6        Args:
7            theme (TerminalTheme, optional): Optional terminal theme, or None to use default. Defaults to None.

9

11            ColorTriplet: A color triplet containing RGB components.
12        """
13
14        if theme is None:
15            theme = DEFAULT_TERMINAL_THEME
16        if self.type == ColorType.TRUECOLOR:
17            assert self.triplet is not None
18            return self.triplet
19        elif self.type == ColorType.EIGHT_BIT:
20            assert self.number is not None
21            return EIGHT_BIT_PALETTE[self.number]
22        elif self.type == ColorType.STANDARD:
23            assert self.number is not None
24            return theme.ansi_colors[self.number]
25        elif self.type == ColorType.WINDOWS:
26            assert self.number is not None
27            return WINDOWS_PALETTE[self.number]
28        else:  # self.type == ColorType.DEFAULT:
29            assert self.number is None
30            return theme.foreground_color if foreground else theme.background_color
