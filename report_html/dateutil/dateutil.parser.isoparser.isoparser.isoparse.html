2    def isoparse(self, dt_str):
3        """
4        Parse an ISO-8601 datetime string into a :class:`datetime.datetime`.
5
6        An ISO-8601 datetime string consists of a date portion, followed
7        optionally by a time portion - the date and time portions are separated
8        by a single character separator, which is ``T`` in the official
9        standard. Incomplete date formats (such as ``YYYY-MM``) may *not* be
10        combined with a time portion.
11
12        Supported date formats are:
13
14        Common:
15
16        - ``YYYY``
17        - ``YYYY-MM``
18        - ``YYYY-MM-DD`` or ``YYYYMMDD``
19
20        Uncommon:
21
22        - ``YYYY-Www`` or ``YYYYWww`` - ISO week (day defaults to 0)
23        - ``YYYY-Www-D`` or ``YYYYWwwD`` - ISO week and day
24
25        The ISO week and day numbering follows the same logic as
26        :func:`datetime.date.isocalendar`.
27
28        Supported time formats are:
29
30        - ``hh``
31        - ``hh:mm`` or ``hhmm``
32        - ``hh:mm:ss`` or ``hhmmss``
33        - ``hh:mm:ss.ssssss`` (Up to 6 sub-second digits)
34
35        Midnight is a special case for `hh`, as the standard supports both
36        00:00 and 24:00 as a representation. The decimal separator can be
37        either a dot or a comma.
38
39
40        .. caution::
41
42            Support for fractional components other than seconds is part of the
43            ISO-8601 standard, but is not currently implemented in this parser.
44
45        Supported time zone offset formats are:
46
47        - `Z` (UTC)
48        - `±HH:MM`
49        - `±HHMM`
50        - `±HH`
51
52        Offsets will be represented as :class:`dateutil.tz.tzoffset` objects,
53        with the exception of UTC, which will be represented as
54        :class:`dateutil.tz.tzutc`. Time zone offsets equivalent to UTC (such
55        as `+00:00`) will also be represented as :class:`dateutil.tz.tzutc`.
56
57        :param dt_str:
58            A string or stream containing only an ISO-8601 datetime string
59
60        :return:
61            Returns a :class:`datetime.datetime` representing the string.
62            Unspecified components default to their lowest value.
63
64        .. warning::
65
66            As of version 2.7.0, the strictness of the parser should not be
67            considered a stable part of the contract. Any valid ISO-8601 string
68            that parses correctly with the default settings will continue to
69            parse correctly in future versions, but invalid strings that
70            currently fail (e.g. ``2017-01-01T00:00+00:00:00``) are not
71            guaranteed to continue failing in future versions if they encode
72            a valid date.
73
74        .. versionadded:: 2.7.0
75        """
76        components, pos = self._parse_isodate(dt_str)
77
78        if len(dt_str) > pos:
79            if self._sep is None or dt_str[pos:pos + 1] == self._sep:
80                components += self._parse_isotime(dt_str[pos + 1:])
81            else:
82                raise ValueError('String contains unknown ISO components')
83
84        if len(components) > 3 and components[3] == 24:
85            components[3] = 0
86            return datetime(*components) + timedelta(days=1)
87
88        return datetime(*components)
