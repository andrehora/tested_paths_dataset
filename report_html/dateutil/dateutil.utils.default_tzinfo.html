2    """
3    Sets the ``tzinfo`` parameter on naive datetimes only
4
5    This is useful for example when you are provided a datetime that may have
6    either an implicit or explicit time zone, such as when parsing a time zone

8

10

12        >>> from dateutil.parser import parse
13        >>> from dateutil.utils import default_tzinfo
14        >>> dflt_tz = tzoffset("EST", -18000)
15        >>> print(default_tzinfo(parse('2014-01-01 12:30 UTC'), dflt_tz))
16        2014-01-01 12:30:00+00:00
17        >>> print(default_tzinfo(parse('2014-01-01 12:30'), dflt_tz))
18        2014-01-01 12:30:00-05:00
19
20    :param dt:
21        The datetime on which to replace the time zone
22
23    :param tzinfo:
24        The :py:class:`datetime.tzinfo` subclass instance to assign to
25        ``dt`` if (and only if) it is naive.
26
27    :return:
28        Returns an aware :py:class:`datetime.datetime`.
29    """
30    if dt.tzinfo is not None:
31        return dt
32    else:
33        return dt.replace(tzinfo=tzinfo)
