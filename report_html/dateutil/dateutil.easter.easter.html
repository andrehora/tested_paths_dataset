1def easter(year, method=EASTER_WESTERN):
2    """
3    This method was ported from the work done by GM Arts,
4    on top of the algorithm by Claus Tondering, which was
5    based in part on the algorithm of Ouding (1940), as
6    quoted in "Explanatory Supplement to the Astronomical
7    Almanac", P.  Kenneth Seidelmann, editor.
8
9    This algorithm implements three different Easter
10    calculation methods:
11
12    1. Original calculation in Julian calendar, valid in
13       dates after 326 AD
14    2. Original method, with date converted to Gregorian
15       calendar, valid in years 1583 to 4099
16    3. Revised method, in Gregorian calendar, valid in
17       years 1583 to 4099 as well
18
19    These methods are represented by the constants:
20
21    * ``EASTER_JULIAN   = 1``
22    * ``EASTER_ORTHODOX = 2``
23    * ``EASTER_WESTERN  = 3``
24
25    The default method is method 3.
26
27    More about the algorithm may be found at:
28
29    `GM Arts: Easter Algorithms <http://www.gmarts.org/index.php?go=415>`_
30
31    and
32
33    `The Calendar FAQ: Easter <https://www.tondering.dk/claus/cal/easter.php>`_
34
35    """
36
37    if not (1 <= method <= 3):
38        raise ValueError("invalid method")
39
40    # g - Golden year - 1
41    # c - Century
42    # h - (23 - Epact) mod 30
43    # i - Number of days from March 21 to Paschal Full Moon
44    # j - Weekday for PFM (0=Sunday, etc)
45    # p - Number of days from March 21 to Sunday on or before PFM
46    #     (-6 to 28 methods 1 & 3, to 56 for method 2)
47    # e - Extra days to add for method 2 (converting Julian
48    #     date to Gregorian date)
49
50    y = year
51    g = y % 19
52    e = 0
53    if method < 3:
54        # Old method
55        i = (19*g + 15) % 30
56        j = (y + y//4 + i) % 7
57        if method == 2:
58            # Extra dates to convert Julian to Gregorian date
59            e = 10
60            if y > 1600:
61                e = e + y//100 - 16 - (y//100 - 16)//4
62    else:
63        # New method
64        c = y//100
65        h = (c - c//4 - (8*c + 13)//25 + 19*g + 15) % 30
66        i = h - (h//28)*(1 - (h//28)*(29//(h + 1))*((21 - g)//11))
67        j = (y + y//4 + i + 2 - c + c//4) % 7
68
69    # p can be from -6 to 56 corresponding to dates 22 March to 23 May
70    # (later dates apply to method 2, although 23 May never actually occurs)
71    p = i - j + e
72    d = 1 + (p + 27 + (p + 6)//40) % 31
73    m = 3 + (p + 26)//30
74    return datetime.date(int(y), int(m), int(d))
