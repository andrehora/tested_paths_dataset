41                components[comp] = int(us_str) * 10**(6 - len(us_str))
42                pos += len(frac.group())
43
44        if pos < len_str:
45            raise ValueError('Unused components in ISO string')
46
47        if components[0] == 24:
48            # Standard supports 00:00 and 24:00 as representations of midnight
49            if any(component != 0 for component in components[1:4]):
50                raise ValueError('Hour may only be 24 at 24:00:00.000')
51
52        return components
