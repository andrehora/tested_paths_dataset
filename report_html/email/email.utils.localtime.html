1def localtime(dt=None, isdst=-1):
2    """Return local time as an aware datetime object.
3
4    If called without arguments, return current time.  Otherwise *dt*
5    argument should be a datetime instance, and it is converted to the
6    local time zone according to the system time zone database.  If *dt* is
7    naive (that is, dt.tzinfo is None), it is assumed to be in local time.
8    In this case, a positive or zero value for *isdst* causes localtime to
9    presume initially that summer time (for example, Daylight Saving Time)
10    is or is not (respectively) in effect for the specified time.  A
11    negative value for *isdst* causes the localtime() function to attempt
12    to divine whether summer time is in effect for the specified time.
13
14    """
15    if dt is None:
16        return datetime.datetime.now(datetime.timezone.utc).astimezone()
17    if dt.tzinfo is not None:
18        return dt.astimezone()
19    # We have a naive datetime.  Convert to a (localtime) timetuple and pass to
20    # system mktime together with the isdst hint.  System mktime will return
21    # seconds since epoch.
22    tm = dt.timetuple()[:-1] + (isdst,)
23    seconds = time.mktime(tm)
24    localtm = time.localtime(seconds)
25    try:
26        delta = datetime.timedelta(seconds=localtm.tm_gmtoff)
27        tz = datetime.timezone(delta, localtm.tm_zone)
28    except AttributeError:
29        # Compute UTC offset and compare with the value implied by tm_isdst.
30        # If the values match, use the zone name implied by tm_isdst.
31        delta = dt - datetime.datetime(*time.gmtime(seconds)[:6])
32        dst = time.daylight and localtm.tm_isdst > 0
33        gmtoff = -(time.altzone if dst else time.timezone)
34        if delta == datetime.timedelta(seconds=gmtoff):
35            tz = datetime.timezone(delta, time.tzname[dst])
36        else:
37            tz = datetime.timezone(delta)
38    return dt.replace(tzinfo=tz)
