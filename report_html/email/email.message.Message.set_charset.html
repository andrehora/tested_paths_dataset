1def set_charset(self, charset):
2        """Set the charset of the payload to a given character set.
3
4        charset can be a Charset instance, a string naming a character set, or
5        None.  If it is a string it will be converted to a Charset instance.
6        If charset is None, the charset parameter will be removed from the
7        Content-Type field.  Anything else will generate a TypeError.
8
9        The message will be assumed to be of type text/* encoded with
10        charset.input_charset.  It will be converted to charset.output_charset
11        and encoded properly, if needed, when generating the plain text
12        representation of the message.  MIME headers (MIME-Version,
13        Content-Type, Content-Transfer-Encoding) will be added as needed.
14        """
15        if charset is None:
16            self.del_param('charset')
17            self._charset = None
18            return
19        if not isinstance(charset, Charset):
20            charset = Charset(charset)
21        self._charset = charset
22        if 'MIME-Version' not in self:
23            self.add_header('MIME-Version', '1.0')
24        if 'Content-Type' not in self:
25            self.add_header('Content-Type', 'text/plain',
26                            charset=charset.get_output_charset())
27        else:
28            self.set_param('charset', charset.get_output_charset())
29        if charset != charset.get_output_charset():
30            self._payload = charset.body_encode(self._payload)
31        if 'Content-Transfer-Encoding' not in self:
32            cte = charset.get_body_encoding()
33            try:
34                cte(self)
35            except TypeError:
36                # This 'if' is for backward compatibility, it allows unicode
37                # through even though that won't work correctly if the
38                # message is serialized.
39                payload = self._payload
40                if payload:
41                    try:
42                        payload = payload.encode('ascii', 'surrogateescape')
43                    except UnicodeError:
44                        payload = payload.encode(charset.output_charset)
45                self._payload = charset.body_encode(payload)
46                self.add_header('Content-Transfer-Encoding', cte)
