1def set_type(self, type, header='Content-Type', requote=True):
2        """Set the main type and subtype for the Content-Type header.
3
4        type must be a string in the form "maintype/subtype", otherwise a
5        ValueError is raised.
6
7        This method replaces the Content-Type header, keeping all the
8        parameters in place.  If requote is False, this leaves the existing
9        header's quoting as is.  Otherwise, the parameters will be quoted (the
10        default).
11
12        An alternative header can be specified in the header argument.  When
13        the Content-Type header is set, we'll always also add a MIME-Version
14        header.
15        """
16        # BAW: should we be strict?
17        if not type.count('/') == 1:
18            raise ValueError
19        # Set the Content-Type, you get a MIME-Version
20        if header.lower() == 'content-type':
21            del self['mime-version']
22            self['MIME-Version'] = '1.0'
23        if header not in self:
24            self[header] = type
25            return
26        params = self.get_params(header=header, unquote=requote)
27        del self[header]
28        self[header] = type
29        # Skip the first param; it's the old type.
30        for p, v in params[1:]:
31            self.set_param(p, v, header, requote)
