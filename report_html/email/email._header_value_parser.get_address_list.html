1def get_address_list(value):
2    """ address_list = (address *("," address)) / obs-addr-list
3        obs-addr-list = *([CFWS] ",") address *("," [address / CFWS])
4
5    We depart from the formal grammar here by continuing to parse until the end
6    of the input, assuming the input to be entirely composed of an
7    address-list.  This is always true in email parsing, and allows us
8    to skip invalid addresses to parse additional valid ones.
9
10    """
11    address_list = AddressList()
12    while value:
13        try:
14            token, value = get_address(value)
15            address_list.append(token)
16        except errors.HeaderParseError as err:
17            leader = None
18            if value[0] in CFWS_LEADER:
19                leader, value = get_cfws(value)
20                if not value or value[0] == ',':
21                    address_list.append(leader)
22                    address_list.defects.append(errors.ObsoleteHeaderDefect(
23                        "address-list entry with no content"))
24                else:
25                    token, value = get_invalid_mailbox(value, ',')
26                    if leader is not None:
27                        token[:0] = [leader]
28                    address_list.append(Address([token]))
29                    address_list.defects.append(errors.InvalidHeaderDefect(
30                        "invalid address in address-list"))
31            elif value[0] == ',':
32                address_list.defects.append(errors.ObsoleteHeaderDefect(
33                    "empty element in address-list"))
34            else:
35                token, value = get_invalid_mailbox(value, ',')
36                if leader is not None:
37                    token[:0] = [leader]
38                address_list.append(Address([token]))
39                address_list.defects.append(errors.InvalidHeaderDefect(
40                    "invalid address in address-list"))
41        if value and value[0] != ',':
42            # Crap after address; treat it as an invalid mailbox.
43            # The mailbox info will still be available.
44            mailbox = address_list[-1][0]
45            mailbox.token_type = 'invalid-mailbox'
46            token, value = get_invalid_mailbox(value, ',')
47            mailbox.extend(token)
48            address_list.defects.append(errors.InvalidHeaderDefect(
49                "invalid address in address-list"))
50        if value:  # Must be a , at this point.
51            address_list.append(ValueTerminal(',', 'list-separator'))
52            value = value[1:]
53    return address_list, value
