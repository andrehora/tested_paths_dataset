1def _handle_multipart(self, msg):
2        # The trick here is to write out each part separately, merge them all
3        # together, and then make sure that the boundary we've chosen isn't
4        # present in the payload.
5        msgtexts = []
6        subparts = msg.get_payload()
7        if subparts is None:
8            subparts = []
9        elif isinstance(subparts, str):
10            # e.g. a non-strict parse of a message with no starting boundary.
11            self.write(subparts)
12            return
13        elif not isinstance(subparts, list):
14            # Scalar payload
15            subparts = [subparts]
16        for part in subparts:
17            s = self._new_buffer()
18            g = self.clone(s)
19            g.flatten(part, unixfrom=False, linesep=self._NL)
20            msgtexts.append(s.getvalue())
21        # BAW: What about boundaries that are wrapped in double-quotes?
22        boundary = msg.get_boundary()
23        if not boundary:
24            # Create a boundary that doesn't appear in any of the
25            # message texts.
26            alltext = self._encoded_NL.join(msgtexts)
27            boundary = self._make_boundary(alltext)
28            msg.set_boundary(boundary)
29        # If there's a preamble, write it out, with a trailing CRLF
30        if msg.preamble is not None:
31            if self._mangle_from_:
32                preamble = fcre.sub('>From ', msg.preamble)
33            else:
34                preamble = msg.preamble
35            self._write_lines(preamble)
36            self.write(self._NL)
37        # dash-boundary transport-padding CRLF
38        self.write('--' + boundary + self._NL)
39        # body-part
40        if msgtexts:
41            self._fp.write(msgtexts.pop(0))
42        # *encapsulation
43        # --> delimiter transport-padding
44        # --> CRLF body-part
45        for body_part in msgtexts:
46            # delimiter transport-padding CRLF
47            self.write(self._NL + '--' + boundary + self._NL)
48            # body-part
49            self._fp.write(body_part)
50        # close-delimiter transport-padding
51        self.write(self._NL + '--' + boundary + '--' + self._NL)
52        if msg.epilogue is not None:
53            if self._mangle_from_:
54                epilogue = fcre.sub('>From ', msg.epilogue)
55            else:
56                epilogue = msg.epilogue
57            self._write_lines(epilogue)
