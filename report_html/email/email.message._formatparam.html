7    be encoded according to RFC2231 rules, using the utf-8 charset and
8    a null language.
9    """
10    if value is not None and len(value) > 0:

12        # are (charset, language, value).  charset is a string, not a Charset
13        # instance.  RFC 2231 encoded values are never quoted, per RFC.
14        if isinstance(value, tuple):
15            # Encode as per RFC 2231
16            param += '*'
17            value = utils.encode_rfc2231(value[2], value[0], value[1])
18            return '%s=%s' % (param, value)
19        else:
20            try:
21                value.encode('ascii')
22            except UnicodeEncodeError:
23                param += '*'

25                return '%s=%s' % (param, value)
