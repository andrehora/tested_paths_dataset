1def parse_mime_parameters(value):
2    """ parameter *( ";" parameter )
3
4    That BNF is meant to indicate this routine should only be called after
5    finding and handling the leading ';'.  There is no corresponding rule in
6    the formal RFC grammar, but it is more convenient for us for the set of
7    parameters to be treated as its own TokenList.
8
9    This is 'parse' routine because it consumes the remaining value, but it
10    would never be called to parse a full header.  Instead it is called to
11    parse everything after the non-parameter value of a specific MIME header.
12
13    """
14    mime_parameters = MimeParameters()
15    while value:
16        try:
17            token, value = get_parameter(value)
18            mime_parameters.append(token)
19        except errors.HeaderParseError as err:
20            leader = None
21            if value[0] in CFWS_LEADER:
22                leader, value = get_cfws(value)
23            if not value:
24                mime_parameters.append(leader)
25                return mime_parameters
26            if value[0] == ';':
27                if leader is not None:
28                    mime_parameters.append(leader)
29                mime_parameters.defects.append(errors.InvalidHeaderDefect(
30                    "parameter entry with no content"))
31            else:
32                token, value = get_invalid_parameter(value)
33                if leader:
34                    token[:0] = [leader]
35                mime_parameters.append(token)
36                mime_parameters.defects.append(errors.InvalidHeaderDefect(
37                    "invalid parameter {!r}".format(token)))
38        if value and value[0] != ';':
39            # Junk after the otherwise valid parameter.  Mark it as
40            # invalid, but it will have a value.
41            param = mime_parameters[-1]
42            param.token_type = 'invalid-parameter'
43            token, value = get_invalid_parameter(value)
44            param.extend(token)
45            mime_parameters.defects.append(errors.InvalidHeaderDefect(
46                "parameter with invalid trailing text {!r}".format(token)))
47        if value:
48            # Must be a ';' at this point.
49            mime_parameters.append(ValueTerminal(';', 'parameter-separator'))
50            value = value[1:]
51    return mime_parameters
