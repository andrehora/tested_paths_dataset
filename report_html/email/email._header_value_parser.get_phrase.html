1def get_phrase(value):
2    """ phrase = 1*word / obs-phrase
3        obs-phrase = word *(word / "." / CFWS)
4
5    This means a phrase can be a sequence of words, periods, and CFWS in any
6    order as long as it starts with at least one word.  If anything other than
7    words is detected, an ObsoleteHeaderDefect is added to the token's defect
8    list.  We also accept a phrase that starts with CFWS followed by a dot;
9    this is registered as an InvalidHeaderDefect, since it is not supported by
10    even the obsolete grammar.
11
12    """
13    phrase = Phrase()
14    try:
15        token, value = get_word(value)
16        phrase.append(token)
17    except errors.HeaderParseError:
18        phrase.defects.append(errors.InvalidHeaderDefect(
19            "phrase does not start with word"))
20    while value and value[0] not in PHRASE_ENDS:
21        if value[0]=='.':
22            phrase.append(DOT)
23            phrase.defects.append(errors.ObsoleteHeaderDefect(
24                "period in 'phrase'"))
25            value = value[1:]
26        else:
27            try:
28                token, value = get_word(value)
29            except errors.HeaderParseError:
30                if value[0] in CFWS_LEADER:
31                    token, value = get_cfws(value)
32                    phrase.defects.append(errors.ObsoleteHeaderDefect(
33                        "comment found without atom"))
34                else:
35                    raise
36            phrase.append(token)
37    return phrase, value
