1def decode_params(params):
2    """Decode parameters list according to RFC 2231.
3
4    params is a sequence of 2-tuples containing (param name, string value).
5    """
6    new_params = [params[0]]
7    # Map parameter's name to a list of continuations.  The values are a
8    # 3-tuple of the continuation number, the string value, and a flag
9    # specifying whether a particular segment is %-encoded.
10    rfc2231_params = {}
11    for name, value in params[1:]:
12        encoded = name.endswith('*')
13        value = unquote(value)
14        mo = rfc2231_continuation.match(name)
15        if mo:
16            name, num = mo.group('name', 'num')
17            if num is not None:
18                num = int(num)
19            rfc2231_params.setdefault(name, []).append((num, value, encoded))
20        else:
21            new_params.append((name, '"%s"' % quote(value)))
22    if rfc2231_params:
23        for name, continuations in rfc2231_params.items():
24            value = []
25            extended = False
26            # Sort by number
27            continuations.sort()
28            # And now append all values in numerical order, converting
29            # %-encodings for the encoded segments.  If any of the
30            # continuation names ends in a *, then the entire string, after
31            # decoding segments and concatenating, must have the charset and
32            # language specifiers at the beginning of the string.
33            for num, s, encoded in continuations:
34                if encoded:
35                    # Decode as "latin-1", so the characters in s directly
36                    # represent the percent-encoded octet values.
37                    # collapse_rfc2231_value treats this as an octet sequence.
38                    s = urllib.parse.unquote(s, encoding="latin-1")
39                    extended = True
40                value.append(s)
41            value = quote(EMPTYSTRING.join(value))
42            if extended:
43                charset, language, value = decode_rfc2231(value)
44                new_params.append((name, (charset, language, '"%s"' % value)))
45            else:
46                new_params.append((name, '"%s"' % value))
47    return new_params
