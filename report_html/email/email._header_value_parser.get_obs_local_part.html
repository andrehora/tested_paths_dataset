1def get_obs_local_part(value):

22            continue

26        try:
27            token, value = get_word(value)
28            last_non_ws_was_dot = False
29        except errors.HeaderParseError:

32            token, value = get_cfws(value)
33        obs_local_part.append(token)
34    if (obs_local_part[0].token_type == 'dot' or
35            obs_local_part[0].token_type=='cfws' and
36            obs_local_part[1].token_type=='dot'):
37        obs_local_part.defects.append(errors.InvalidHeaderDefect(
38            "Invalid leading '.' in local part"))
39    if (obs_local_part[-1].token_type == 'dot' or
40            obs_local_part[-1].token_type=='cfws' and
41            obs_local_part[-2].token_type=='dot'):
42        obs_local_part.defects.append(errors.InvalidHeaderDefect(
43            "Invalid trailing '.' in local part"))
44    if obs_local_part.defects:

46    return obs_local_part, value
