1def decode_header(header):
2    """Decode a message header value without converting charset.
3
4    Returns a list of (string, charset) pairs containing each of the decoded
5    parts of the header.  Charset is None for non-encoded parts of the header,
6    otherwise a lower-case string containing the name of the character set
7    specified in the encoded string.
8
9    header may be a string that may or may not contain RFC2047 encoded words,
10    or it may be a Header object.
11
12    An email.errors.HeaderParseError may be raised when certain decoding error
13    occurs (e.g. a base64 decoding exception).
14    """
15    # If it is a Header object, we can just return the encoded chunks.
16    if hasattr(header, '_chunks'):
17        return [(_charset._encode(string, str(charset)), str(charset))
18                    for string, charset in header._chunks]
19    # If no encoding, just return the header with no charset.
20    if not ecre.search(header):
21        return [(header, None)]
22    # First step is to parse all the encoded parts into triplets of the form
23    # (encoded_string, encoding, charset).  For unencoded strings, the last
24    # two parts will be None.
25    words = []
26    for line in header.splitlines():
27        parts = ecre.split(line)
28        first = True
29        while parts:
30            unencoded = parts.pop(0)
31            if first:
32                unencoded = unencoded.lstrip()
33                first = False
34            if unencoded:
35                words.append((unencoded, None, None))
36            if parts:
37                charset = parts.pop(0).lower()
38                encoding = parts.pop(0).lower()
39                encoded = parts.pop(0)
40                words.append((encoded, encoding, charset))
41    # Now loop over words and remove words that consist of whitespace
42    # between two encoded strings.
43    droplist = []
44    for n, w in enumerate(words):
45        if n>1 and w[1] and words[n-2][1] and words[n-1][0].isspace():
46            droplist.append(n-1)
47    for d in reversed(droplist):
48        del words[d]
49
50    # The next step is to decode each encoded word by applying the reverse
51    # base64 or quopri transformation.  decoded_words is now a list of the
52    # form (decoded_word, charset).
53    decoded_words = []
54    for encoded_string, encoding, charset in words:
55        if encoding is None:
56            # This is an unencoded word.
57            decoded_words.append((encoded_string, charset))
58        elif encoding == 'q':

60            decoded_words.append((word, charset))
61        elif encoding == 'b':

63            if paderr:
64                encoded_string += '==='[:4 - paderr]
65            try:

67            except binascii.Error:
68                raise HeaderParseError('Base64 decoding error')
69            else:
70                decoded_words.append((word, charset))
71        else:
72            raise AssertionError('Unexpected encoding: ' + encoding)
73    # Now convert all words to bytes and collapse consecutive runs of
74    # similarly encoded words.
75    collapsed = []
76    last_word = last_charset = None
77    for word, charset in decoded_words:
78        if isinstance(word, str):
79            word = bytes(word, 'raw-unicode-escape')
80        if last_word is None:
81            last_word = word

83        elif charset != last_charset:
84            collapsed.append((last_word, last_charset))
85            last_word = word
86            last_charset = charset
87        elif last_charset is None:
88            last_word += BSPACE + word
89        else:
90            last_word += word
91    collapsed.append((last_word, last_charset))
92    return collapsed
