1def flatten(self, msg, unixfrom=False, linesep=None):
2        r"""Print the message object tree rooted at msg to the output file
3        specified when the Generator instance was created.
4
5        unixfrom is a flag that forces the printing of a Unix From_ delimiter
6        before the first object in the message tree.  If the original message
7        has no From_ delimiter, a `standard' one is crafted.  By default, this
8        is False to inhibit the printing of any From_ delimiter.
9
10        Note that for subobjects, no From_ line is printed.
11
12        linesep specifies the characters used to indicate a new line in
13        the output.  The default value is determined by the policy specified
14        when the Generator instance was created or, if none was specified,
15        from the policy associated with the msg.
16
17        """
18        # We use the _XXX constants for operating on data that comes directly
19        # from the msg, and _encoded_XXX constants for operating on data that
20        # has already been converted (to bytes in the BytesGenerator) and
21        # inserted into a temporary buffer.
22        policy = msg.policy if self.policy is None else self.policy
23        if linesep is not None:
24            policy = policy.clone(linesep=linesep)
25        if self.maxheaderlen is not None:
26            policy = policy.clone(max_line_length=self.maxheaderlen)
27        self._NL = policy.linesep
28        self._encoded_NL = self._encode(self._NL)
29        self._EMPTY = ''
30        self._encoded_EMPTY = self._encode(self._EMPTY)
31        # Because we use clone (below) when we recursively process message
32        # subparts, and because clone uses the computed policy (not None),
33        # submessages will automatically get set to the computed policy when
34        # they are processed by this code.
35        old_gen_policy = self.policy
36        old_msg_policy = msg.policy
37        try:
38            self.policy = policy
39            msg.policy = policy
40            if unixfrom:
41                ufrom = msg.get_unixfrom()
42                if not ufrom:
43                    ufrom = 'From nobody ' + time.ctime(time.time())
44                self.write(ufrom + self._NL)
45            self._write(msg)
46        finally:
47            self.policy = old_gen_policy
48            msg.policy = old_msg_policy
