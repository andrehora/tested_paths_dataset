1def get_unstructured(value):
2    """unstructured = (*([FWS] vchar) *WSP) / obs-unstruct
3       obs-unstruct = *((*LF *CR *(obs-utext) *LF *CR)) / FWS)
4       obs-utext = %d0 / obs-NO-WS-CTL / LF / CR
5
6       obs-NO-WS-CTL is control characters except WSP/CR/LF.
7
8    So, basically, we have printable runs, plus control characters or nulls in
9    the obsolete syntax, separated by whitespace.  Since RFC 2047 uses the
10    obsolete syntax in its specification, but requires whitespace on either
11    side of the encoded words, I can see no reason to need to separate the
12    non-printable-non-whitespace from the printable runs if they occur, so we
13    parse this into xtext tokens separated by WSP tokens.
14
15    Because an 'unstructured' value must by definition constitute the entire
16    value, this 'get' routine does not return a remaining value, only the
17    parsed TokenList.
18
19    """
20    # XXX: but what about bare CR and LF?  They might signal the start or
21    # end of an encoded word.  YAGNI for now, since our current parsers
22    # will never send us strings with bare CR or LF.
23
24    unstructured = UnstructuredTokenList()
25    while value:
26        if value[0] in WSP:
27            token, value = get_fws(value)
28            unstructured.append(token)
29            continue
30        valid_ew = True
31        if value.startswith('=?'):
32            try:
33                token, value = get_encoded_word(value)
34            except _InvalidEwError:
35                valid_ew = False
36            except errors.HeaderParseError:
37                # XXX: Need to figure out how to register defects when
38                # appropriate here.
39                pass
40            else:
41                have_ws = True
42                if len(unstructured) > 0:
43                    if unstructured[-1].token_type != 'fws':
44                        unstructured.defects.append(errors.InvalidHeaderDefect(
45                            "missing whitespace before encoded word"))
46                        have_ws = False
47                if have_ws and len(unstructured) > 1:
48                    if unstructured[-2].token_type == 'encoded-word':
49                        unstructured[-1] = EWWhiteSpaceTerminal(
50                            unstructured[-1], 'fws')
51                unstructured.append(token)
52                continue
53        tok, *remainder = _wsp_splitter(value, 1)
54        # Split in the middle of an atom if there is a rfc2047 encoded word
55        # which does not have WSP on both sides. The defect will be registered
56        # the next time through the loop.
57        # This needs to only be performed when the encoded word is valid;
58        # otherwise, performing it on an invalid encoded word can cause
59        # the parser to go in an infinite loop.
60        if valid_ew and rfc2047_matcher.search(tok):
61            tok, *remainder = value.partition('=?')
62        vtext = ValueTerminal(tok, 'vtext')
63        _validate_xtext(vtext)
64        unstructured.append(vtext)
65        value = ''.join(remainder)
66    return unstructured

1def get_unstructured(value):
2    """unstructured = (*([FWS] vchar) *WSP) / obs-unstruct
3       obs-unstruct = *((*LF *CR *(obs-utext) *LF *CR)) / FWS)
4       obs-utext = %d0 / obs-NO-WS-CTL / LF / CR
5
6       obs-NO-WS-CTL is control characters except WSP/CR/LF.
7
8    So, basically, we have printable runs, plus control characters or nulls in
9    the obsolete syntax, separated by whitespace.  Since RFC 2047 uses the
10    obsolete syntax in its specification, but requires whitespace on either
11    side of the encoded words, I can see no reason to need to separate the
12    non-printable-non-whitespace from the printable runs if they occur, so we
13    parse this into xtext tokens separated by WSP tokens.
14
15    Because an 'unstructured' value must by definition constitute the entire
16    value, this 'get' routine does not return a remaining value, only the
17    parsed TokenList.
18
19    """
20    # XXX: but what about bare CR and LF?  They might signal the start or
21    # end of an encoded word.  YAGNI for now, since our current parsers
22    # will never send us strings with bare CR or LF.
23
24    unstructured = UnstructuredTokenList()
25    while value:
26        if value[0] in WSP:
27            token, value = get_fws(value)
28            unstructured.append(token)
29            continue
30        valid_ew = True
31        if value.startswith('=?'):
32            try:
33                token, value = get_encoded_word(value)
34            except _InvalidEwError:
35                valid_ew = False
36            except errors.HeaderParseError:
37                # XXX: Need to figure out how to register defects when
38                # appropriate here.
39                pass
40            else:
41                have_ws = True
42                if len(unstructured) > 0:
43                    if unstructured[-1].token_type != 'fws':
44                        unstructured.defects.append(errors.InvalidHeaderDefect(
45                            "missing whitespace before encoded word"))
46                        have_ws = False
47                if have_ws and len(unstructured) > 1:
48                    if unstructured[-2].token_type == 'encoded-word':
49                        unstructured[-1] = EWWhiteSpaceTerminal(
50                            unstructured[-1], 'fws')
51                unstructured.append(token)
52                continue
53        tok, *remainder = _wsp_splitter(value, 1)
54        # Split in the middle of an atom if there is a rfc2047 encoded word
55        # which does not have WSP on both sides. The defect will be registered
56        # the next time through the loop.
57        # This needs to only be performed when the encoded word is valid;
58        # otherwise, performing it on an invalid encoded word can cause
59        # the parser to go in an infinite loop.
60        if valid_ew and rfc2047_matcher.search(tok):
61            tok, *remainder = value.partition('=?')
62        vtext = ValueTerminal(tok, 'vtext')
63        _validate_xtext(vtext)
64        unstructured.append(vtext)
65        value = ''.join(remainder)
66    return unstructured
