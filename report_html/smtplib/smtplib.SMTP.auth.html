1def auth(self, mechanism, authobject, *, initial_response_ok=True):
2        """Authentication command - requires response processing.
3
4        'mechanism' specifies which authentication mechanism is to
5        be used - the valid values are those listed in the 'auth'
6        element of 'esmtp_features'.
7
8        'authobject' must be a callable object taking a single argument:
9
10                data = authobject(challenge)
11
12        It will be called to process the server's challenge response; the
13        challenge argument it is passed will be a bytes.  It should return
14        an ASCII string that will be base64 encoded and sent to the server.
15
16        Keyword arguments:
17            - initial_response_ok: Allow sending the RFC 4954 initial-response
18              to the AUTH command, if the authentication methods supports it.
19        """
20        # RFC 4954 allows auth methods to provide an initial response.  Not all
21        # methods support it.  By definition, if they return something other
22        # than None when challenge is None, then they do.  See issue #15014.
23        mechanism = mechanism.upper()
24        initial_response = (authobject() if initial_response_ok else None)
25        if initial_response is not None:
26            response = encode_base64(initial_response.encode('ascii'), eol='')
27            (code, resp) = self.docmd("AUTH", mechanism + " " + response)
28            self._auth_challenge_count = 1
29        else:
30            (code, resp) = self.docmd("AUTH", mechanism)
31            self._auth_challenge_count = 0
32        # If server responds with a challenge, send the response.
33        while code == 334:
34            self._auth_challenge_count += 1
35            challenge = base64.decodebytes(resp)
36            response = encode_base64(
37                authobject(challenge).encode('ascii'), eol='')
38            (code, resp) = self.docmd(response)
39            # If server keeps sending challenges, something is wrong.
40            if self._auth_challenge_count > _MAXCHALLENGE:
41                raise SMTPException(
42                    "Server AUTH mechanism infinite loop. Last response: "
43                    + repr((code, resp))
44                )
45        if code in (235, 503):
46            return (code, resp)
47        raise SMTPAuthenticationError(code, resp)
