1def send_message(self, msg, from_addr=None, to_addrs=None,
2                     mail_options=(), rcpt_options=()):
3        """Converts message to a bytestring and passes it to sendmail.
4
5        The arguments are as for sendmail, except that msg is an
6        email.message.Message object.  If from_addr is None or to_addrs is
7        None, these arguments are taken from the headers of the Message as
8        described in RFC 2822 (a ValueError is raised if there is more than
9        one set of 'Resent-' headers).  Regardless of the values of from_addr and
10        to_addr, any Bcc field (or Resent-Bcc field, when the Message is a
11        resent) of the Message object won't be transmitted.  The Message
12        object is then serialized using email.generator.BytesGenerator and
13        sendmail is called to transmit the message.  If the sender or any of
14        the recipient addresses contain non-ASCII and the server advertises the
15        SMTPUTF8 capability, the policy is cloned with utf8 set to True for the
16        serialization, and SMTPUTF8 and BODY=8BITMIME are asserted on the send.
17        If the server does not support SMTPUTF8, an SMTPNotSupported error is
18        raised.  Otherwise the generator is called without modifying the
19        policy.
20
21        """
22        # 'Resent-Date' is a mandatory field if the Message is resent (RFC 2822
23        # Section 3.6.6). In such a case, we use the 'Resent-*' fields.  However,
24        # if there is more than one 'Resent-' block there's no way to
25        # unambiguously determine which one is the most recent in all cases,
26        # so rather than guess we raise a ValueError in that case.
27        #
28        # TODO implement heuristics to guess the correct Resent-* block with an
29        # option allowing the user to enable the heuristics.  (It should be
30        # possible to guess correctly almost all of the time.)
31
32        self.ehlo_or_helo_if_needed()
33        resent = msg.get_all('Resent-Date')
34        if resent is None:
35            header_prefix = ''
36        elif len(resent) == 1:
37            header_prefix = 'Resent-'
38        else:
39            raise ValueError("message has more than one 'Resent-' header block")
40        if from_addr is None:
41            # Prefer the sender field per RFC 2822:3.6.2.
42            from_addr = (msg[header_prefix + 'Sender']
43                           if (header_prefix + 'Sender') in msg
44                           else msg[header_prefix + 'From'])
45            from_addr = email.utils.getaddresses([from_addr])[0][1]
46        if to_addrs is None:
47            addr_fields = [f for f in (msg[header_prefix + 'To'],
48                                       msg[header_prefix + 'Bcc'],
49                                       msg[header_prefix + 'Cc'])
50                           if f is not None]
51            to_addrs = [a[1] for a in email.utils.getaddresses(addr_fields)]
52        # Make a local copy so we can delete the bcc headers.
53        msg_copy = copy.copy(msg)
54        del msg_copy['Bcc']
55        del msg_copy['Resent-Bcc']
56        international = False
57        try:
58            ''.join([from_addr, *to_addrs]).encode('ascii')
59        except UnicodeEncodeError:
60            if not self.has_extn('smtputf8'):
61                raise SMTPNotSupportedError(
62                    "One or more source or delivery addresses require"
63                    " internationalized email support, but the server"
64                    " does not advertise the required SMTPUTF8 capability")
65            international = True
66        with io.BytesIO() as bytesmsg:
67            if international:
68                g = email.generator.BytesGenerator(
69                    bytesmsg, policy=msg.policy.clone(utf8=True))
70                mail_options = (*mail_options, 'SMTPUTF8', 'BODY=8BITMIME')
71            else:
72                g = email.generator.BytesGenerator(bytesmsg)
73            g.flatten(msg_copy, linesep='\r\n')
74            flatmsg = bytesmsg.getvalue()
75        return self.sendmail(from_addr, to_addrs, flatmsg, mail_options,
76                             rcpt_options)

25        # unambiguously determine which one is the most recent in all cases,
26        # so rather than guess we raise a ValueError in that case.
27        #
28        # TODO implement heuristics to guess the correct Resent-* block with an
29        # option allowing the user to enable the heuristics.  (It should be
30        # possible to guess correctly almost all of the time.)
31
32        self.ehlo_or_helo_if_needed()
33        resent = msg.get_all('Resent-Date')
34        if resent is None:
35            header_prefix = ''
36        elif len(resent) == 1:
37            header_prefix = 'Resent-'
38        else:
39            raise ValueError("message has more than one 'Resent-' header block")
40        if from_addr is None:
41            # Prefer the sender field per RFC 2822:3.6.2.
42            from_addr = (msg[header_prefix + 'Sender']
43                           if (header_prefix + 'Sender') in msg
44                           else msg[header_prefix + 'From'])
45            from_addr = email.utils.getaddresses([from_addr])[0][1]
46        if to_addrs is None:
47            addr_fields = [f for f in (msg[header_prefix + 'To'],
48                                       msg[header_prefix + 'Bcc'],
49                                       msg[header_prefix + 'Cc'])
50                           if f is not None]
51            to_addrs = [a[1] for a in email.utils.getaddresses(addr_fields)]
52        # Make a local copy so we can delete the bcc headers.
53        msg_copy = copy.copy(msg)
54        del msg_copy['Bcc']
55        del msg_copy['Resent-Bcc']
56        international = False
57        try:
58            ''.join([from_addr, *to_addrs]).encode('ascii')
59        except UnicodeEncodeError:
60            if not self.has_extn('smtputf8'):
61                raise SMTPNotSupportedError(
62                    "One or more source or delivery addresses require"
63                    " internationalized email support, but the server"
64                    " does not advertise the required SMTPUTF8 capability")
65            international = True
66        with io.BytesIO() as bytesmsg:
67            if international:
68                g = email.generator.BytesGenerator(
69                    bytesmsg, policy=msg.policy.clone(utf8=True))
70                mail_options = (*mail_options, 'SMTPUTF8', 'BODY=8BITMIME')
71            else:
72                g = email.generator.BytesGenerator(bytesmsg)
73            g.flatten(msg_copy, linesep='\r\n')
74            flatmsg = bytesmsg.getvalue()
75        return self.sendmail(from_addr, to_addrs, flatmsg, mail_options,
76                             rcpt_options)
