6        while rest:
7            p = rest.find("%")
8            if p < 0:
9                accum.append(rest)
10                return
11            if p > 0:
12                accum.append(rest[:p])
13                rest = rest[p:]
14            # p is no longer used
15            c = rest[1:2]
16            if c == "%":
17                accum.append("%")
18                rest = rest[2:]
19            elif c == "(":
20                m = self._KEYCRE.match(rest)
21                if m is None:
22                    raise InterpolationSyntaxError(option, section,
23                        "bad interpolation variable reference %r" % rest)
24                var = parser.optionxform(m.group(1))
25                rest = rest[m.end():]
26                try:
27                    v = map[var]
28                except KeyError:
29                    raise InterpolationMissingOptionError(
30                        option, section, rawval, var) from None
31                if "%" in v:
32                    self._interpolate_some(parser, option, accum, v,
33                                           section, map, depth + 1)
34                else:
35                    accum.append(v)
36            else:
37                raise InterpolationSyntaxError(
38                    option, section,
39                    "'%%' must be followed by '%%' or '(', "
40                    "found: %r" % (rest,))
