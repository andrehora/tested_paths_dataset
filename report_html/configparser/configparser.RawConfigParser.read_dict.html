1def read_dict(self, dictionary, source='<dict>'):
2        """Read configuration from a dictionary.
3
4        Keys are section names, values are dictionaries with keys and values
5        that should be present in the section. If the used dictionary type
6        preserves order, sections and their keys will be added in order.
7
8        All types held in the dictionary are converted to strings during
9        reading, including section names, option names and keys.
10
11        Optional second argument is the `source' specifying the name of the
12        dictionary being read.
13        """
14        elements_added = set()
15        for section, keys in dictionary.items():
16            section = str(section)
17            try:
18                self.add_section(section)
19            except (DuplicateSectionError, ValueError):
20                if self._strict and section in elements_added:
21                    raise
22            elements_added.add(section)
23            for key, value in keys.items():
24                key = self.optionxform(str(key))
25                if value is not None:
26                    value = str(value)
27                if self._strict and (section, key) in elements_added:
28                    raise DuplicateOptionError(section, key, source)
29                elements_added.add((section, key))
30                self.set(section, key, value)
