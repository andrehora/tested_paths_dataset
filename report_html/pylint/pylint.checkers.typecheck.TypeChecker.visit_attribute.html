39            and self.linter.config.ignore_on_opaque_inference
40        ):
41            # There is an ambiguity in the inference. Since we can't
42            # make sure that we won't emit a false positive, we just stop
43            # whenever the inference returns an opaque inference object.
44            return
45        for owner in non_opaque_inference_results:
46            name = getattr(owner, "name", None)
47            if _is_owner_ignored(
48                owner,
49                name,
50                self.linter.config.ignored_classes,
51                self.linter.config.ignored_modules,
52            ):
53                continue
54
55            qualname = f"{owner.pytype()}.{node.attrname}"
56            if any(
57                pattern.match(qualname) for pattern in self._compiled_generated_members
58            ):
59                return
60
61            try:
62                attr_nodes = owner.getattr(node.attrname)
63            except AttributeError:
64                continue
65            except astroid.DuplicateBasesError:
66                continue
67            except astroid.NotFoundError:
68                # This can't be moved before the actual .getattr call,
69                # because there can be more values inferred and we are
70                # stopping after the first one which has the attribute in question.
71                # The problem is that if the first one has the attribute,
72                # but we continue to the next values which doesn't have the
73                # attribute, then we'll have a false positive.
74                # So call this only after the call has been made.
75                if not _emit_no_member(
76                    node,
77                    owner,
78                    name,
79                    self._mixin_class_rgx,
80                    ignored_mixins=(
81                        "no-member" in self.linter.config.ignored_checks_for_mixins
82                    ),
83                    ignored_none=self.linter.config.ignore_none,
84                ):
85                    continue
86                missingattr.add((owner, name))
87                continue
88            else:
89                for attr_node in attr_nodes:
90                    attr_parent = attr_node.parent
91                    # Skip augmented assignments
92                    try:
93                        if isinstance(
94                            attr_node.statement(future=True), nodes.AugAssign
95                        ) or (
96                            isinstance(attr_parent, nodes.Assign)
97                            and utils.is_augmented_assign(attr_parent)[0]
98                        ):
99                            continue
100                    except astroid.exceptions.StatementMissing:
101                        break
102                    # Skip self-referencing assignments
103                    if attr_parent is node.parent:
104                        continue
105                    break
106                else:
107                    missingattr.add((owner, name))
108                    continue
109            # stop on the first found
110            break
111        else:
112            # we have not found any node with the attributes, display the
113            # message for inferred nodes
114            done = set()
115            for owner, name in missingattr:
116                if isinstance(owner, astroid.Instance):
117                    actual = owner._proxied
118                else:
119                    actual = owner
120                if actual in done:
121                    continue
122                done.add(actual)
123
124                msg, hint = self._get_nomember_msgid_hint(node, owner)
125                self.add_message(
126                    msg,
127                    node=node,
128                    args=(owner.display_type(), name, node.attrname, hint),
129                    confidence=INFERENCE,
130                )
