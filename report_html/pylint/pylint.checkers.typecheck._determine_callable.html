22        return callable_obj, parameters, "lambda"
23    if isinstance(callable_obj, nodes.ClassDef):
24        # Class instantiation, lookup __new__ instead.
25        # If we only find object.__new__, we can safely check __init__
26        # instead. If __new__ belongs to builtins, then we look
27        # again for __init__ in the locals, since we won't have
28        # argument information for the builtin __new__ function.
29        try:
30            # Use the last definition of __new__.
31            new = callable_obj.local_attr("__new__")[-1]
32        except astroid.NotFoundError:
33            new = None
34
35        from_object = new and new.parent.scope().name == "object"
36        from_builtins = new and new.root().name in sys.builtin_module_names
37
38        if not new or from_object or from_builtins:
39            try:
40                # Use the last definition of __init__.
41                callable_obj = callable_obj.local_attr("__init__")[-1]
42            except astroid.NotFoundError as e:
43                raise ValueError from e
44        else:
45            callable_obj = new
46
47        if not isinstance(callable_obj, nodes.FunctionDef):
48            raise ValueError
49        # both have an extra implicit 'cls'/'self' argument.
50        return callable_obj, parameters, "constructor"
51
52    raise ValueError
