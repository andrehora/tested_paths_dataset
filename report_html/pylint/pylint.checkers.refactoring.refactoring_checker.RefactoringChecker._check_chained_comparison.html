15        ) -> None:

19                    value = None
20                    if isinstance(operand, nodes.Name):
21                        value = operand.name
22                    elif isinstance(operand, nodes.Const):
23                        value = operand.value
24

26                        continue
27
28                    if operator in {"<", "<="}:
29                        if operand is left_operand:
30                            uses[value]["lower_bound"].add(comparison_node)
31                        elif operand is right_operand:
32                            uses[value]["upper_bound"].add(comparison_node)
33                    elif operator in {">", ">="}:
34                        if operand is left_operand:
35                            uses[value]["upper_bound"].add(comparison_node)

37                            uses[value]["lower_bound"].add(comparison_node)
38                left_operand = right_operand
39
40        uses: collections.defaultdict[

42        ] = collections.defaultdict(
43            lambda: {"lower_bound": set(), "upper_bound": set()}
44        )

47                _find_lower_upper_bounds(comparison_node, uses)
48
