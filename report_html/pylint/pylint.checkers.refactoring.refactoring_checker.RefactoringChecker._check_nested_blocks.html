16            for ancestor_node in reversed(self._nested_blocks):
17                if ancestor_node == node.parent:
18                    break
19                self._nested_blocks.pop()
20            # if the node is an elif, this should not be another nesting level
21            if isinstance(node, nodes.If) and self._is_actual_elif(node):
22                if self._nested_blocks:
23                    self._nested_blocks.pop()
24            self._nested_blocks.append(node)
25
