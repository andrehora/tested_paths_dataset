1def _no_context_variadic(
2    node: nodes.Call,
3    variadic_name: str | None,
4    variadic_type: nodes.Keyword | nodes.Starred,
5    variadics: list[nodes.Keyword | nodes.Starred],
6) -> bool:
7    """Verify if the given call node has variadic nodes without context.
8
9    This is a workaround for handling cases of nested call functions
10    which don't have the specific call context at hand.
11    Variadic arguments (variable positional arguments and variable
12    keyword arguments) are inferred, inherently wrong, by astroid
13    as a Tuple, respectively a Dict with empty elements.
14    This can lead pylint to believe that a function call receives
15    too few arguments.
16    """
17    scope = node.scope()
18    is_in_lambda_scope = not isinstance(scope, nodes.FunctionDef) and isinstance(
19        scope, nodes.Lambda
20    )
21    statement = node.statement(future=True)
22    for name in statement.nodes_of_class(nodes.Name):
23        if name.name != variadic_name:
24            continue
25
26        inferred = safe_infer(name)
27        if isinstance(inferred, (nodes.List, nodes.Tuple)):
28            length = len(inferred.elts)
29        elif isinstance(inferred, nodes.Dict):
30            length = len(inferred.items)
31        else:
32            continue
33
34        if is_in_lambda_scope and isinstance(inferred.parent, nodes.Arguments):
35            # The statement of the variadic will be the assignment itself,
36            # so we need to go the lambda instead
37            inferred_statement = inferred.parent.parent
38        else:
39            inferred_statement = inferred.statement(future=True)
40
41        if not length and isinstance(inferred_statement, nodes.Lambda):
42            is_in_starred_context = _has_parent_of_type(node, variadic_type, statement)
43            used_as_starred_argument = any(
44                variadic.value == name or variadic.value.parent_of(name)
45                for variadic in variadics
46            )
47            if is_in_starred_context or used_as_starred_argument:
48                return True
49    return False
