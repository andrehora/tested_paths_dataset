2        self,
3        node: nodes.Call,
4        call_site: arguments.CallSite,
5        called: CallableObjects,
6        called_param_names: list[str | None],
7    ) -> None:
8        """Match the supplied argument names against the function parameters.
9
10        Warn if some argument names are not in the same order as they are in
11        the function signature.
12        """
13        # Check for called function being an object instance function
14        # If so, ignore the initial 'self' argument in the signature
15        try:
16            is_classdef = isinstance(called.parent, nodes.ClassDef)
17            if is_classdef and called_param_names[0] == "self":
18                called_param_names = called_param_names[1:]
19        except IndexError:
20            return
21
22        try:
23            # extract argument names, if they have names
24            calling_parg_names = [p.name for p in call_site.positional_arguments]
25
26            # Additionally, get names of keyword arguments to use in a full match
27            # against parameters
28            calling_kwarg_names = [
29                arg.name for arg in call_site.keyword_arguments.values()
30            ]
31        except AttributeError:
32            # the type of arg does not provide a `.name`. In this case we
33            # stop checking for out-of-order arguments because it is only relevant
34            # for named variables.
35            return
36
37        # Don't check for ordering if there is an unmatched arg or param
38        arg_set = set(calling_parg_names) | set(calling_kwarg_names)
39        param_set = set(called_param_names)
40        if arg_set != param_set:
41            return
42
43        # Warn based on the equality of argument ordering
44        if calling_parg_names != called_param_names[: len(calling_parg_names)]:
45            self.add_message("arguments-out-of-order", node=node, args=())
