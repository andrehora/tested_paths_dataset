30            ):
31                self.add_message("bad-staticmethod-argument", args=first, node=node)
32                return

37        # class / regular method with no args
38        elif not (
39            node.args.args
40            or node.args.posonlyargs
41            or node.args.vararg
42            or node.args.kwarg
43        ):
44            self.add_message("no-method-argument", node=node, args=node.name)
45        # metaclass
46        elif metaclass:
47            # metaclass __new__ or classmethod
48            if node.type == "classmethod":

56            # metaclass regular method
57            else:

63                    node.name,

72                node.name,

74        # regular class with regular method without self as argument
